===
navigate to admin page
%require
===
TOKEN=$(cat "$CCTR_FIXTURE_DIR/gateway-token.txt")
WORKER_URL=$(cat "$CCTR_FIXTURE_DIR/worker-url.txt")
plwr -S cinaworker-e2e open "$WORKER_URL/_admin/?token=$TOKEN" -T 120000
---

===
wait for pending devices and approve
%require
===
plwr -S cinaworker-e2e wait 'button:has-text("Approve All")' -T 120000
plwr -S cinaworker-e2e click 'button:has-text("Approve All")' -T 10000
# Wait for the approval API call to finish before navigating away.
# The button changes to "Approving..." then disappears once complete.
plwr -S cinaworker-e2e wait-not 'button:has-text("Approv")' -T 30000
---

===
navigate to chat page
%require
===
TOKEN=$(cat "$CCTR_FIXTURE_DIR/gateway-token.txt")
WORKER_URL=$(cat "$CCTR_FIXTURE_DIR/worker-url.txt")
plwr -S cinaworker-e2e open "$WORKER_URL/?token=$TOKEN" -T 120000

===
wait for chat interface to load
%require
===
plwr -S cinaworker-e2e wait 'textarea:not([disabled])' -T 120000
---

===
send math question
%require
===
plwr -S cinaworker-e2e fill textarea 'What is 847293 + 651824? Reply with just the number.'
plwr -S cinaworker-e2e press Enter
---

===
wait for correct answer
===
plwr -S cinaworker-e2e wait 'text=1499117' -T 120000
